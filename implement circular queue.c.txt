#include <stdio.h>
#define N 5

int q[N], front = -1, rear = -1;

void enqueue(int x) {
    if ((rear + 1) % N == front) {
        printf("Overflow\n");
        return;
    }
    if (front == -1) front = 0;
    rear = (rear + 1) % N;
    q[rear] = x;
    printf("Inserted: %d\n", x);
}

void dequeue() {
    if (front == -1) {
        printf("Underflow\n");
        return;
    }
    printf("Deleted: %d\n", q[front]);
    if (front == rear)
        front = rear = -1;
    else
        front = (front + 1) % N;
}

void display() {
    if (front == -1) {
        printf("Queue Empty\n");
        return;
    }
    int i = front;
    printf("Queue: ");
    while (1) {
        printf("%d ", q[i]);
        if (i == rear) break;
        i = (i + 1) % N;
    }
    printf("\n");
}

int main() {
    enqueue(10);
    enqueue(20);
    enqueue(30);
    dequeue();
    enqueue(40);
    display();
    return 0;
}
